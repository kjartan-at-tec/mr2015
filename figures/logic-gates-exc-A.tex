\documentclass[]{standalone}
\usepackage[utf8]{inputenc}
\usepackage[american]{circuitikz}
\usetikzlibrary{arrows,shapes,calc,positioning,circuits.logic.US,circuits.logic.IEC}

\begin{document}

\pgfmathsetmacro\circuitheight{6}
\pgfmathsetmacro\circuitwidth{7}

\begin{circuitikz}[scale=1]
  \node[and port] (and1) at (0,0) {};
  \node[nand port] (nand1) at (3,-0.3) {};
  \node[and port] (and2) at (0.1,-2) {};
  \node[or port] (or1) at (6,-1) {};

  \draw
  (and1.out) -- node[coordinate, pos=0.6] (branch) {} (nand1.in 1)
  (branch) |- (nand1.in 2)
  (nand1.out) -| (or1.in 1)
  (and2.out) -| (or1.in 2)
  (and1.in 2) ++(0.1,0) -- (and2.in 1)
  (and1.in 1) node[anchor=east] {$x$}
  (and1.in 2) node[anchor=east] {$y$}
  (and2.in 2) node[anchor=east] {$z$};
\end{circuitikz}
\end{document}
